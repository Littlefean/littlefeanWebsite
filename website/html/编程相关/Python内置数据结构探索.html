小顶堆

  1. import heapq
  3. h = [i for i in range(10)]
  4. h.reverse()
  5. heapq.heapify(h)  # 这个函数没有返回值，直接传递列表改变列表
  6. print(h)
  7. for i in range(10):
  8.     print(heapq.heappop(h))  # 每次弹出的时候都会维护一下小顶
  9.     print(h)
