  1. class Solution:
  2.     def maxSubArray(self, nums: List[int]) -> int:
  3.         dp = [0] * len(nums)
  4.         dp[0] = nums[0]
  5.         res = nums[0]
  6.         for i, n in enumerate(nums):
  7.             if i == 0:
  8.                 continue
  9.             if dp[i - 1] <= 0:
  10.                 # 如果从前面到现在已经小于0了，或者等于零了，前面的部分就不要要了
  11.                 dp[i] = nums[i]
  12.             else:
  13.                 dp[i] = dp[i - 1] + nums[i]
  14.             res = max(res, dp[i])
  15.         return res

python递归缓存

来自竞赛题 https://leetcode-cn.com/problems/longest-path-with-different-adjacent-characters/

在dfs函数上面加@lru_cache(None) 能通过

  1. class Solution:
  2.     def longestPath(self, parent: List[int], s: str) -> int:
  3.         d = {}
  4.         for i, n in enumerate(parent):
  5.             if n not in d:
  6.                 d[n] = set()
  7.             if i not in d:
  8.                 d[i] = set()
  9.             d[n].add(i)
  10.             d[i].add(n)
  11.         @lru_cache(None)
  12.         def dfs(i, last):#当前节点和上一节点
  13.             if i == -1:
  14.                 return 0
  15.             res = 0
  16.             for j in d[i]:
  17.                 if s[j] != s[i] and j != last:#字符不同，且上下连通节点不同
  18.                     res = max(res, dfs(j, i))
  19.             return res + 1
  20.         res = 0
  21.         for k in range(len(parent)):
  22.             res = max(res, dfs(k, None))
  23.         return res
