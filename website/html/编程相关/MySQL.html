select...from

  1. SELECT 1 + 1, 3 * 2;

  1. SELECT employee_id emp_id FROM employees;

  1. SELECT employee_id AS emp_id FROM employees;

  1. SELECT employee_id AS "emp_id" FROM employees;

  1. SELECT DISTINCT employee_id FROM employees;

分页

一次显示太多的记录，看不过来

  1. SELECT empolyee_id, last_name
  2. FROM empolyees
  3. LIMIT 0, 20;  # 偏移量，往后走多少个
  5. SELECT empolyee_id, last_name
  6. FROM empolyees
  7. LIMIT 20, 20;  # 从第20个开始，往后走20个
  9. # 8.0 新特性
  10. SELECT empolyee_id, last_name
  11. FROM empolyees
  12. LIMIT 20 OFFSET 0;  # 从第1个开始，往后走20个

公式：

  1. SELECT empolyee_id, last_name
  2. FROM empolyees
  3. LIMIT (pageNo - 1) * pageSize, pageSize;  # 每一页显示pageSize条数据，此时显示pageNo页

多表查询

为什么要分成很多表？

  • 数据冗余小
  • 硬盘IO小，一次性不用加载太多
  • 多个人可以同时对不同的表查询，并发高
/*
员工(...., 部门Id)

部门(...., 部门Id, 城市id)

城市(..., 城市id)
*/

# 查询员工
SELECT empolyee_id, department_name
FROM empolyees, departments
WHERE employees.`department_id` = departments.department_id;  # 连接多表

# 如果查询语句中出现了多个表中都存在的字段，必须指明这个字段在的表
# 建议：sql优化的角度，每个字段前面都加上表名
SELECT empolyee_id, department_name, empolyees.department_id
FROM empolyees, departments
WHERE employees.`department_id` = departments.department_id;  # 连接多表

# 一旦使用了别名，就不要再用原来的名字了

# 员工，城市
SELECT empolyee_id, last_name, department_name, city,
FROM empolyees e, departments d, locations l
WHERE e.`department_id` = d.`department_id`
AND d.`location_id` = l.`location_id`;
